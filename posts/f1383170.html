def predict(self, X,y,theta):
    m=len(y)
    X=np.array(X)
    X=np.hstack((np.ones((m,1)),X))  
    p=np.dot(X,theta)
    p=p.reshape(-1,1)
    y=np.array(y).reshape(-1,1)
    sum=0
    for i in range(m):
        if y[i]==1 and p[i]>0:
        	sum+=1
        elif y[i]==0 and p[i]<0:
        	sum+=1
    return sum/m
